Combined Pine Script (paste in Pine Editor):
//@version=5
indicator("Richmond Signal",overlay=true)
rsi = ta.rsi(close,14)
e9 = ta.ema(close,9)
e21 = ta.ema(close,21)
longSig = ta.crossover(e9,e21) and rsi < 55
shortSig = ta.crossunder(e9,e21) and rsi > 45
alertcondition(longSig,"LONG","Bull cross + RSI OK")
alertcondition(shortSig,"SHORT","Bear cross + RSI OK")
